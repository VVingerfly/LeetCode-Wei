18  vector<int> twoSum(vector<int>& nums, int target)
19  {
20  vector<int> res;
21  res.resize(2, 0);
22  unordered_map<int, int> numsHash;
23  for (size_t i = 0; i < nums.size(); ++i)
24  {
25  int value1 = nums.at(i);
26  auto it = numsHash.find(target - value1);
27  if (it != numsHash.end())
28  {
29  res[0] = i < it->second ? i : it->second;
30  res[1] = i < it->second ? it->second : i;
31  break;
32  }
33  numsHash.insert(std::make_pair(nums.at(i), i)); // if not found, store the <value, index> in hash for later searching
34  }
35  return res;
36  }
37 };
