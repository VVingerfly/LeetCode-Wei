54  for (cntCols = 0; cntCols < s.length(); ++cntCols) {
55  sortStr.push_back(make_pair(cntRows, cntCols));
56 
57  goDown ? cntRows++ : cntRows--;
58  if (goDown && cntRows == numRows) {
59  cntRows--;
60  cntRows--;
61  goDown = false;
62  continue;
63  }
64  if (!goDown && cntRows == -1) {
65  cntRows++;
66  cntRows++;
67  goDown = true;
68  continue;
69  }
70  }
71 
72  // printSortStr(sortStr, s);
73  string sOut;
74  for (int row = 0; row < numRows; ++row) {
75  for (int i = 0; i < sortStr.size(); ++i) {
76  if (sortStr.at(i).first == row) {
77  sOut += s.at(sortStr.at(i).second);
78  }
